var bettinghost = require("./bettinghost"),
	readline = require('readline'),
	lineReader;
 
startLineReader();
 
readLines();
 
function readLines(){
	lineReader.on('line', function(line){
		
		var tokens = bettinghost.tokenizeLine(line);
 
		if(bettinghost.processInputTokens(tokens) === "Result"){
			lineReader.close();
			if(!computeResults())
				console.log("Error in computing results. Check the README for instructions");
		}
	});
}
 
function startLineReader(){
	lineReader = readline.createInterface({
		input: process.stdin,
		output: process.stdout,
		terminal: false
	});
}
 
function computeResults(){
	if (bettinghost.computeResults())
		return true;
	else
		return false;
}
